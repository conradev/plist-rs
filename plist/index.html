Parsing a Plist from a file

extern crate plist;

use std::fs::File;
use std::io::BufReader;

use plist::Plist;

fn main() {
    let mut f = File::open("tests/types-xml.plist").unwrap();
    let mut reader = BufReader::new(f);

    let plist = Plist::from_reader(&mut reader).unwrap();

    println!("Property list {:?}", plist);
}
